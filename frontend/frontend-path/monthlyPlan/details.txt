// 封装fetch请求
function request(url, method = 'GET') {
  return new Promise((resolve, reject) => {
    fetch(url, { method })
      .then(response => {
        if (!response.ok) throw new Error('请求失败')
        return response.json()
      })
      .then(resolve)
      .catch(reject)
  })
}
